20  /**
21   * The explorer option allows users to pan and zoom Google charts. explorer: {} provides the default explorer behavior,
22   * enabling users to pan horizontally and vertically by dragging, and to zoom in and out by scrolling.
23   * 
24   * This feature is experimental and may change in future releases.
25   */
26  public class Explorer extends JavaScriptObject {
27  	/**
28  	 * Default constructor.
29  	 * 
30  	 * @return a new object instance
31  	 */
32  	public static Explorer create() {
33  		return createObject().cast();
34  	}
35  
36  	protected Explorer() {
37  	}
38  
39  	/**
40  	 * Sets available actions.<br>
41  	 * We recommend using {@link ExplorerAction#RIGHT_CLICK_TO_RESET} whenever {@link ExplorerAction#DRAG_TO_ZOOM} is
42  	 * used.<br>
43  	 * See {@link #setMaxZoomIn(double)}, {@link #setMaxZoomOut(double)}, {@link #setZoomDelta(double)}, for zoom
44  	 * customizations.
45  	 * 
46  	 * @param actions
47  	 */
48  	public final void setActions(String... actions) {
49  		setActions(ArrayHelper.createArray(actions));
50  	}
51  
52  	/**
53  	 * Limits explorer panning to horizontal or vertical.
54  	 * By default, users can pan both horizontally and vertically when the explorer option is used.
55  	 * 
56  	 * @param explorerAxis
57  	 */
58  	public final native void setAxis(ExplorerAxis explorerAxis) /*-{
59  		this.axis = panning;
60  	}-*/;
61  
62  	/**
63  	 * By default, users can pan all around, regardless of where the data is. To ensure that users don't pan beyond the
64  	 * original chart set this to true.
65  	 * 
66  	 * @param keepInBounds
67  	 */
68  	public final native void setKeepInBounds(boolean keepInBounds) /*-{
69  		this.keepInBounds = keepInBounds;
70  	}-*/;
71  
72  	/**
73  	 * The maximum that the explorer can zoom in. By default, users will be able to zoom in enough that they'll see only
74  	 * 25% of the original view. Setting 0.5 would let users zoom in only far enough to see half of the original view.
75  	 * 
76  	 * @param maxZoomIn default is 0.25
77  	 */
78  	public final native void setMaxZoomIn(double maxZoomIn) /*-{
79  		this.maxZoomIn = maxZoomIn;
80  	}-*/;
81  
82  	/**
83  	 * The maximum that the explorer can zoom out. By default, users will be able to zoom out far enough that the chart
84  	 * will take up only 1/4 of the available space. Setting to 8 would let users zoom out far
85  	 * enough that the chart would take up only 1/8 of the available space.
86  	 * 
87  	 * @param maxZoomOut default is 4
88  	 */
89  	public final native void setMaxZoomOut(double maxZoomOut) /*-{
90  		this.maxZoomOut = maxZoomOut;
91  	}-*/;
92  
93  	/**
94  	 * When users zoom in or out determines how much they zoom by. The smaller the number, the
95  	 * smoother and slower the zoom.
96  	 * 
97  	 * @param zoomDelta default is 1.5
98  	 */
99  	public final native void setZoomDelta(double zoomDelta) /*-{
100 		this.zoomDelta = zoomDelta;
101 	}-*/;
102 
103 	private final native void setActions(JsArrayString actions) /*-{
104 		this.actions = actions;
105 	}-*/;
106 }
