21  /**
22   * An object with members to configure various horizontal axis elements.
23   */
24  public class HAxis extends Axis {

32  	}
33  
34  	/**
35  	 * Constructor with title.
36  	 * 
37  	 * @param title axis title
38  	 * 
39  	 * @return a new object instance
40  	 */
41  	public static HAxis create(String title) {
42  		HAxis hAxis = createObject().cast();
43  		hAxis.setTitle(title);
44  		return hAxis;
45  	}

47  	protected HAxis() {
48  	}

50  	/**
51  	 * If false, will hide outermost labels rather than allow them to be cropped by the chart container. If true, will
52  	 * allow label cropping.<br>
53  	 * This option is only supported for a discrete axis.
54  	 * 
55  	 * @param allowContainerBoundaryTextCufoff
56  	 */
57  	public final native void setAllowContainerBoundaryTextCufoff(boolean allowContainerBoundaryTextCufoff) /*-{
58  		this.allowContainerBoundaryTextCufoff = allowContainerBoundaryTextCufoff;
59  	}-*/;
60  
61  	/**
62  	 * Maximum number of levels of horizontal axis text. If axis text labels become too crowded, the server might shift
63  	 * neighboring labels up or down in order to fit labels closer together. This value specifies the most number of
64  	 * levels to use; the server can use fewer levels, if labels can fit without overlapping.<br>
65  	 * This option is only supported for a discrete axis.
66  	 * 
67  	 * @param levels
68  	 */
69  	public final native void setMaxAlternation(int levels) /*-{
70  		this.maxAlternation = levels;
71  	}-*/;
72  
73  	/**
74  	 * Maximum number of lines allowed for the text labels. Labels can span multiple lines if they are too long, and the
75  	 * nuber of lines is, by default, limited by the height of the available space.<br>
76  	 * This option is only supported for a discrete axis.
77  	 * 
78  	 * @param maxTextLines
79  	 */
80  	public final native void setMaxTextLines(int maxTextLines) /*-{
81  		this.maxTextLines = maxTextLines;
82  	}-*/;
83  
84  	/**
85  	 * Moves the max value of the horizontal axis to the specified value; this will be rightward in most charts. Ignored
86  	 * if this is set to a value smaller than the maximum x-value of the data. hAxis.viewWindow.max overrides this
87  	 * property.<br>
88  	 * This option is only supported for a continuous axis.
89  	 * 
90  	 * @param maxValue
91  	 */
92  	public final void setMaxValue(Date maxValue) {
93  		setMaxValueDate(DateHelper.getJsDate(maxValue));
94  	}
95  
96  	/**
97  	 * Minimum horizontal spacing, in pixels, allowed between two adjacent text labels. If the labels are spaced too
98  	 * densely, or they are too long, the spacing can drop below this threshold, and in this case one of the
99  	 * label-unclutter measures will be applied (e.g, truncating the lables or dropping some of them).<br>
100 	 * This option is only supported for a discrete axis.
101 	 * 
102 	 * @param minTextSpacing
103 	 */
104 	public final native void setMinTextSpacing(int minTextSpacing) /*-{
105 		this.minTextSpacing = minTextSpacing;
106 	}-*/;
107 
108 	/**
109 	 * Moves the min value of the horizontal axis to the specified value; this will be leftward in most charts. Ignored
110 	 * if this is set to a value greater than the minimum x-value of the data. hAxis.viewWindow.min overrides this
111 	 * property.<br>
112 	 * This option is only supported for a continuous axis.
113 	 * 
114 	 * @param minValue
115 	 */
116 	public final void setMinValue(Date minValue) {
117 		setMinValueDate(DateHelper.getJsDate(minValue));
118 	}
119 
120 	/**
121 	 * How many horizontal axis labels to show, where 1 means show every label, 2 means show every other label, and so
122 	 * on. Default is to try to show as many labels as possible without overlapping.<br>
123 	 * This option is only supported for a discrete axis.
124 	 * 
125 	 * @param interval
126 	 */
127 	public final native void setShowTextEvery(int interval) /*-{
128 		this.showTextEvery = interval;
129 	}-*/;
130 
131 	/**
132 	 * If true, draw the horizontal axis text at an angle, to help fit more text along the axis; if false, draw
133 	 * horizontal axis text upright. Default behavior is to slant text if it cannot all fit when drawn upright. Notice
134 	 * that this option is available only when the hAxis.textPosition is set to 'out' (which is the default).<br>
135 	 * This option is only supported for a discrete axis.
136 	 * 
137 	 * @param enabled
138 	 */
139 	public final native void setSlantedText(boolean enabled) /*-{
140 		this.slantedText = enabled;
141 	}-*/;
142 
143 	/**
144 	 * The angle of the horizontal axis text, if it's drawn slanted. Ignored if hAxis.slantedText is false, or is in
145 	 * auto mode, and the chart decided to draw the text horizontally.<br>
146 	 * This option is only supported for a discrete axis.
147 	 * 
148 	 * @param angle
149 	 */
150 	public final native void setSlantedTextAngle(int angle) /*-{
151 		this.slantedTextAngle = angle;
152 	}-*/;

154 	private final native void setMaxValueDate(JsDate maxValue) /*-{
155 		this.maxValue = maxValue;
156 	}-*/;

159 		this.minValue = minValue;
160 	}-*/;
