15  /**
16   * A column role describes the purpose of the data in that column: for example, a column might hold data describing
17   * tooltip text, data point annotations, or uncertainty indicators.
18   * @see <a href="http://developers.google.com/chart/interactive/docs/roles.html">DataTable Roles</a>
19   */
20  public enum RoleType {
21  	/**
22  	 * You should not need to assign this role explicitly unless designing a multi-domain chart (shown here); the basic
23  	 * format of the data table enables the charting engine to infer which columns are domain columns. However, you
24  	 * should be aware of which columns are domain columns so that you know which other columns can modify it. Domain
25  	 * columns specify labels along the major axis of the chart. Multiple domain columns can sometimes be used to
26  	 * support multiple scales within the same chart. Data type: Usually string, but occasionally number or date
27  	 */
28  	DOMAIN("domain"),
29  	/**
30  	 * 
31  	 You should not need to assign this role explicitly; the basic format of the data table enables the charting
32  	 * engine to infer which columns are domain columns. However, you sould be aware of which columns are data columns
33  	 * so that you know which other columns can modify it. Data role columns specify series data to render in the chart.
34  	 * For most charts, one column = one series, but this can vary by chart type (for example, scatter charts require
35  	 * two data columns for the first series, and an additional one for each additional series; candlestick charts
36  	 * require four data columns for each series). Data type: number
37  	 */
38  	DATA("data"),
39  	/**
40  	 * Text to display on the chart near the associated data point. The text displays without any user interaction.
41  	 * Annotations and annotation text can be assigned to both data points and categories (axis labels). There are two
42  	 * styles of annotations: letter (default), which draws the annotation text near the specified point, and line,
43  	 * which draws the annotation text on a line that bisects the chart area. Line annotations only apply to domain
44  	 * values, not data values. You can specify the line style by setting this chart option: annotation: {'column_id':
45  	 * {style: 'line'}} Data type: string Default: Empty string
46  	 */
47  	ANNOTATION("annotation"),
48  	/**
49  	 * Extended text to display when the user hovers over the associated annotation. Annotations and annotation text can
50  	 * be assigned to both data points and categories (axis labels). If you have an annotationText column, you must also
51  	 * have an annotation column. Tooltip text, in contrast, is displayed when the user hovers over the associated data
52  	 * point on the chart. Data type: string Default: Empty string
53  	 */
54  	ANNOTATIONTEXT("annotationText"),
55  	/**
56  	 * Indicates potential data range for a specific point. Intervals are usually displayed as I-bar style range
57  	 * indicators. Interval columns are numeric columns; add interval columns in pairs, marking the low and high value
58  	 * of the bar. Interval values should be added in increasing value, from left to right. Data type: number Default:
59  	 * No interval
60  	 */
61  	INTERVAL("interval"),
62  	/**
63  	 * Text to display when the user hovers over the data point associated with this row. Data type: string Default:
64  	 * Data point value
65  	 */
66  	TOOLTIP("tooltip"),
67  	/**
68  	 * 
69  	 Indicates whether a data point is certain or not. How this is displayed depends on the chart type—for example, it
70  	 * might be indicated by dashed lines or a striped fill. For line and area charts, the segment between two data
71  	 * points is certain if and only if both data points are certain. Data type: boolean, where true is certain, false
72  	 * is uncertain. Default: true
73  	 */
74  	CERTAINTY("certainty"),
75  	/**
76  	 * Emphasizes specified chart data points. Displayed as a thick line and/or large point. For line and area charts,
77  	 * the segment between two data points is emphasized if and only if both data points are emphasized. Data type:
78  	 * boolean Default: false
79  	 */
80  	EMPHASIS("emphasis"),
81  	/**
82  	 * Indicates whether a point is in or out of scope. If a point is out of scope, it is visually de-emphasized. For
83  	 * line and area charts, the segment between two data points is in scope if either endpoint is in scope. Data type:
84  	 * boolean, where true means in scope. Default: true
85  	 */
86  	SCOPE("scope");

88  	/**
89  	 * Get a RoleType by providing its name.
90  	 * 
91  	 * @param name the RoleType name.
92  	 * @return a RoleType corresponding to the provided name.
93  	 */
94  	public static RoleType findByName(String name) {
95  		for (RoleType roleType : RoleType.values()) {
96  			if (roleType.getName().equals(name)) {
97  				return roleType;
98  			}
99  		}
100 		return null;
101 	}

115 	}
