15  /**
16   * Data type of the values of a DataTable column
17   */
18  public enum ColumnType {
19  	STRING("string"), NUMBER("number"), BOOLEAN("boolean"), DATE("date"), DATETIME("datetime"), TIMEOFDAY("timeofday");
20  
21  	/**
22  	 * Get the corresponding ColumnType by name.
23  	 * @param name a string representing the ColumnType name.
24  	 * @return a ColumnType corresponding to the provided name.
25  	 */
26  	public static ColumnType findByName(String name) {
27  		for (ColumnType columnType : ColumnType.values()) {
28  			if (columnType.getName().equals(name)) {
29  				return columnType;
30  			}
31  		}
32  		return null;
33  	}
34  
35  	private final String name;
36  
37  	private ColumnType(String name) {
38  		this.name = name;
39  	}
40  
41  	/**
42  	 * Get the name of the ColumnType.
43  	 * @return the name of the ColumnType.
44  	 */
45  	public String getName() {
46  		return name;
47  	}
48  }
